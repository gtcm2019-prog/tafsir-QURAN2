<script>

/* ───────────── SOURCES ───────────── */
const CDN_DATA  = "https://raw.githubusercontent.com/MohamadHajjRabee/quran-qcf4/main";
const CDN_FONTS = "https://cdn.jsdelivr.net/gh/MohamadHajjRabee/quran-qcf4@main";

const loadedFonts = new Set();

/* ───────────── FONT LOADER ───────────── */
async function loadFont(fontName) {
  if (loadedFonts.has(fontName)) return;

  const file = fontName === "QCF4_QBSML"
    ? `${fontName}.woff2`
    : `${fontName}_W.woff2`;

  const url = `${CDN_FONTS}/fonts-woff2/${file}`;

  const ff = new FontFace(fontName, `url(${url})`);
  await ff.load();
  document.fonts.add(ff);

  loadedFonts.add(fontName);
}

/* ───────────── POPUP ───────────── */
function showPopup(text) {
  const popup = document.getElementById("popup");
  popup.textContent = text;
  popup.style.display = "block";

  setTimeout(() => {
    popup.style.display = "none";
  }, 3000);
}

/* ───────────── LOAD PAGE ───────────── */
async function loadPage(page) {
  const padded = String(page).padStart(3, "0");

  const res = await fetch(`${CDN_DATA}/pages/${padded}.json`);
  const data = await res.json();

  renderPage(data);
}

/* ───────────── RENDER ───────────── */
async function renderPage(data) {
  const container = document.getElementById("page-render");
  container.innerHTML = "";

  const fonts = new Set();
  fonts.add(data.font);

  data.lines.forEach(line => {
    line.words.forEach(word => fonts.add(word.font));
  });

  await Promise.all([...fonts].map(loadFont));

  data.lines.forEach(line => {
    const row = document.createElement("div");
    row.className = "mushaf-line";

    line.words.forEach(word => {
      const span = document.createElement("span");
      span.className = "mushaf-word";
      span.textContent = word.char;
      span.style.fontFamily = word.font;

      /* 🧠 CLICK EVENT */
      span.addEventListener("click", () => {

        // remove old highlight
        document.querySelectorAll(".highlight")
          .forEach(e => e.classList.remove("highlight"));

        // highlight clicked word
        span.classList.add("highlight");

        // show meaning from JSON
        if (word.text) {
          showPopup(
`📖 Mot: ${word.char}
🧠 Sens: ${word.text}`
          );
        } else {
          showPopup("❌ Aucun sens disponible");
        }
      });

      row.appendChild(span);
    });

    container.appendChild(row);
  });
}

/* ───────────── SELECT 1→604 ───────────── */
const sel = document.getElementById("page-select");

for (let i = 1; i <= 604; i++) {
  const opt = document.createElement("option");
  opt.value = i;
  opt.textContent = String(i).padStart(3, "0");
  sel.appendChild(opt);
}

sel.addEventListener("change", () => {
  loadPage(sel.value);
});

/* ───────────── START ───────────── */
loadPage(1);
